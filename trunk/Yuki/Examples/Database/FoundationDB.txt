        private void AppendParameterValue(Stream dest, NpgsqlParameter parameter)
        {
            byte[] serialised;
            if (parameter.DbType == DbType.Binary)
            {
                var ba = (IEnumerable<Byte>)(parameter.Value);
                dest.WriteByte((byte)ASCIIBytes.x);
                dest.WriteByte((byte)ASCIIBytes.SingleQuote);
                foreach (var b in ba)
                {
                    var s = b.ToString("X2");
                    dest.WriteBytes((Byte)(s[0]));
                    dest.WriteBytes((Byte)(s[1]));
                }
                dest.WriteByte((byte)ASCIIBytes.SingleQuote);
            }
            else
            {
                serialised = parameter.TypeInfo.ConvertToBackend(parameter.NpgsqlValue, false, Connector.NativeToBackendTypeConverterOptions);
                dest.WriteBytes(serialised);
            }
            return;
